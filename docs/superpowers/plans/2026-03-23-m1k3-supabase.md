**Goal:** Replace all localStorage state in `public/M1K3/index.html` with Supabase (Google OAuth + real-time sync + CRUD screen for boodschappen and oefeningen).

**Tech Stack:** Supabase JS SDK v2 (CDN), Google OAuth (already configured in Supabase), vanilla JS (no framework), pure HTML/CSS.

```js
  const { data } = await sb.from('user_settings').select('seeded_at').maybeSingle()
  if (data) return  // already seeded

  // Seed boodschappen_items from hardcoded BOODSCHAPPEN
  const bItems = BOODSCHAPPEN.flatMap((cat, ci) =>

    ...PULL.map((ex, i) => ({ user_id: currentUserId, type: 'pull', name: ex.name, sets_count: ex.sets, reps_label: ex.reps, sort_order: i })),
```

- [ ] **Step 3: Add `loadBoodschappenItems()`**

```js
async function loadBoodschappenItems() {
  const { data, error } = await sb.from('boodschappen_items')
    .select('*').order('cat').order('sort_order')
  if (error) throw error

  const catMap = new Map()
  data.forEach(item => {
    if (!catMap.has(item.cat)) catMap.set(item.cat, { cat: item.cat, items: [] })
    catMap.get(item.cat).items.push({
      id: item.id,           // UUID — used as checklist_state.item_id FK
      label: item.label,
      price: parseFloat(item.price) || 0,
    })
  })
  BOODSCHAPPEN.splice(0, BOODSCHAPPEN.length, ...catMap.values())
}
```

```js
async function loadExercises() {

  LEGS.splice(0, LEGS.length, ...legs)
}
```

```js
async function toggleShopItem(row) {
  const itemId = row.dataset.id
  const wy = toWeekYear(new Date())
  if (checkedItemIds.has(itemId)) {
    await sb.from('checklist_state').delete()
      .eq('user_id', currentUserId).eq('item_id', itemId).eq('week_year', wy)
    checkedItemIds.delete(itemId)
  } else {
    await sb.from('checklist_state').insert({ user_id: currentUserId, item_id: itemId, week_year: wy })
    checkedItemIds.add(itemId)
  }
  const isChecked = checkedItemIds.has(itemId)
  row.classList.toggle('checked', isChecked)
  row.setAttribute('aria-checked', isChecked ? 'true' : 'false')
  updateShopUI()
}
```

## Known limitations

- **Exercises indexed by sort_order in sets_data:** If the user deletes an exercise mid-week, the `sets_data` indices shift. Clicking sets on a different day after editing may show wrong checked state for that week. The user should reset fitness state (simply wait for next week) after editing exercises.
- **localStorage active_tab** intentionally kept — purely UX preference, no sensitive data.
